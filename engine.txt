Define Damage(string time)(string enemy def)(string ch)(string user def)(string crit?) (warp=false)
{
    AT = (Operator.Join(Operator.LetterOf(protagonist data[ch], 1), Operator.LetterOf(protagonist data[ch], 2)) * time);
    If (user def == 1)
    {
        AT = Operator.Join(Operator.LetterOf(protagonist data[ch], 3), Operator.LetterOf(protagonist data[ch], 4));
    }
    DE = Operator.Join(Operator.LetterOf(enemy data[enemy], 3), Operator.LetterOf(enemy data[enemy], 4));
    damage = ((Operator.Random(0.8 , 1) * ((AT * AT) / (abs(AT) + DE))) * e ^((protagonist levels[ch] / 8)));
    damage = Operator.Round((0 - abs(damage)));
    If ((Operator.Random(1 , 255) < (Operator.Join(Operator.LetterOf(protagonist data[ch], 5), Operator.LetterOf(protagonist data[ch], 6)) + 1)) Or (crit? == 1))
    {
        Sound.Play(slice);
        damage = ((Operator.Random(-0.8 , -1) * ((AT * AT) / (abs(AT) + 0))) * e ^((protagonist levels[ch] / 8)));
        damage = Operator.Round(damage);
        enemy hp += damage;
        List.Add(battle log, Operator.Join("Critical hit! ", Operator.Join(Enemy Names[enemy], Operator.Join(" took ", Operator.Join((-1 * damage), " damage!")))));
    }
    Else
    {
        enemy hp += damage;
        If (Operator.Random(1 , (11 - (((protagonist status[ch] > 0) And (protagonist status[ch] < 10)) * 9))) == 1)
        {
            enemy hp += (0 - damage);
            damage = 0;
            Sound.Play(Evade);
            List.Add(battle log, Operator.Join(Operator.Join("But ", Enemy Names[enemy]), " dodged the attack!"));
        }
        Else
        {
            List.Add(battle log, Operator.Join(Enemy Names[enemy], Operator.Join(" took ", Operator.Join((-1 * damage), " damage!"))));
        }
    }
}

            Repeat Until ((Sensing.KeyPressed(space) And List.ContainsItem(unlocked skills 1, Hover)) And (Not (((character mp[1] < 1m[Hover]) Or ((Hover == 6) And (pill count == 0))))))
            {
                Call Battle int.(0);
                Call List Attack 1;
            }
            If (Hover == 6)
            {
                Sound.Play(sol);
                pill count += -1;
                Call Heal(1);
                Hover = 100;
            }
            If (Hover == 100)
            {
                List.Add(battle log, Operator.Join(user, " used a pill!"));
                List.Add(battle log, Operator.Join(user, " was fully healed!"));
            }
            Else
            {
                List.Add(battle log, Operator.Join(user, Operator.Join(" used ", Operator.Join(username attacks[Hover], "!"))));
                If (Hover == 2)
                {
                    List.Add(battle log, Operator.Join(Enemy Names[enemy], Operator.Join(" took ", Operator.Join((10 * protagonist levels[1]), " damage!"))));
                }
            }
            If (character mp[1] < 1m[Hover])
            {
                List.Add(battle log, "Insufficient MP!");
            }
            Else
            {
                If (Hover == 1)
                {
                    Sound.Play(attack);
                    Call Damage(-1)(1)(1)("")("");
                }
                If (Hover == 2)
                {
                    Sound.Play(spikes);
                    enemy hp += (-10 * protagonist levels[1]);
                }
                If (Hover == 3)
                {
                    Ironized = 4;
                    Sound.Play(block2);
                    List.Add(battle log, "The team becomes a lump of iron!");
                }
                If (Hover == 4)
                {
                    Sound.Play(Snoring);
                    If (Operator.Random(1 , 3) == 1)
                    {
                        Enemy Sleep = Operator.Random(1 , 2);
                        List.Add(battle log, Operator.Join(Enemy Names[enemy], " falls asleep."));
                    }
                    Else
                    {
                        List.Add(battle log, Operator.Join(Enemy Names[enemy], " did not fall asleep."));
                    }
                }
                If (Hover == 5)
                {
                    Sound.Play(quake);
                    Call Damage(0)(1)(1)(1)("");
                }
                List.ReplaceItem(character mp, 1, (character mp[1] - 1m[Hover]));
            }
        }
    }
}
